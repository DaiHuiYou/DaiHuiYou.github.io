void str_copy(char *from, char *to)  //表明主调函数分配内存，供被调函数内部使用
{
    for( ; *from != '\0'; from++, to++)
    {
        *to = *from;
    }

    *to = '\0';
}

void str_copy1(char *from, char *to)
{
    while( (*to = *from) != '\0')
    {
        to++;
        from++;
    }

}

void str_copy2(char *from, char *to)
{
    while((*to++ = *from++) != '\0')  //等效与*to = *from; to++; from++;
    {
        ;
    }
}

    int reverse(char *str)
{
    int length;
    char *temp = str;
    char *p1;
    char *p2;
    char c;

    if(temp == NULL)
    {
        return -1;
        printf("str = NULL.");
    }
    length = strlen(temp);
    p1 = temp;
    p2 = temp + length -1;

    while(p1 < p2)
    {
        c = *p1;
        *p1 = *p2;
        *p2 = c;
        ++p1;
        --p2;
    }
    return 0;
}

      void SortArray(int a[], int num) // int a[] 数组的形参自动退化成指针。

        }
