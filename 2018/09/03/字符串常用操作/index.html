    字符串COPY函数优化过程

        void str_copy(char *from, char *to)  //表明主调函数分配内存，供被调函数内部使用
    {
        for( ; *from != '\0'; from++, to++)
        {
            *to = *from;
        }

        *to = '\0';
    }

    void str_copy1(char *from, char *to)
    {
        while( (*to = *from) != '\0')
        {
            to++;
            from++;
        }

    }

    void str_copy2(char *from, char *to)
    {
        while((*to++ = *from++) != '\0')  //等效与*to = *from; to++; from++;
        {
            ;
        }
    }

    //建议不要直接修改形参的值， 要引入临时指针变量，把形参接过来
    void str_copy_good(char *from, char *to)
    {
        char *tempfrom = from;
        char *tempto = to;

        if( tempfrom ==NULL || tempto == NULL)
        {
            return;
        }

        while((*tempto++ = *tempfrom++) != '\0');
        printf("to : %s\n",to); //这样可以直接打印，避免COPY后，指针指向\0，打印不出来。
    }

    (count)++和 count++的区别：count++ 等效于 (count++)

    • 输入特性 in：主函数分配内存，供被调函数使用。

    • 输出特性out：被调函数分配内存，供主调函数。

 {

     char * count = ncount;

     {
         ret = -1;

     do
     {
         p = strstr(p, sub);
