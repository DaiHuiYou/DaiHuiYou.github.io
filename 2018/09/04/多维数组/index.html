        printf("hello...\n");
        system("pause");
        return ;
    }

    定义数组指针变量的方法1

    //用数组类型 

        void main()
    {
        char *Myarray[] = {"1111", "33333", "aaaa"}; //指针 数组

        //数组指针  用一个指针 来指向一个数组

        typedef int (MyArrayType)[5]; //定义了一个数据类型  数组数据类型
        int i = 0;
        MyArrayType  myArray;  //int myArray[5]; //用类型定义变量 

        MyArrayType *pArray; //定义一个指针变量 这个指针变量 指向一个数组

        {
            int myArray2[5]; //相当于一级指针

            pArray = &myArray2; //相当于2级指针
            for (i=0; i<5; i++)
            {
                (*pArray)[i] = i+1;
            }

            for (i=0; i<5; i++)
            {
                printf("%d ", (*pArray)[i]);
            }
        }
        printf("hello...\n");
        system("pause");
        return ;

    }

    多维数组作参数退化问问题

        //多维数组作参数的退化问题

    void getArray1(int buff[3][5])//buff退化为数组指针
    {
        int num = 0;
        int i, j;
        int temp = 0;

        num = sizeof(buff);
        printf("num:%d\n", num);

        for(i=0; i<3; i++)
        {
            for(j=0; j<5; j++)
            {
                *(*(buff+i)+j) = temp++;
            }
        }

    }

    void getArray2(int buff[][5])
    {
        int i, j;
        int temp = 0;

        for(i=0; i<3; i++)
        {
            for(j=0; j<5; j++)
            {
                buff[i][j] = temp++;
            }
        }
    }

    void getArray3(int (*buff)[5])
    {
        int i, j;
        int temp = 0;

        for(i=0; i<3; i++)
        {
            for(j=0; j<5; j++)
            {
                *(*(buff+i)+j) = temp++;;
            }
        }
    }

    //多维数组名的本质  数组指针 ，步长 一维的长度

    //
    // (a+i) 代表是整个第i行的地址  二级指针
    // *(a+i) 代表 1级指针  第i行首元素的地址
    // *(a+i) + j  ===> &  a[i][j]

    //*( *(a+i) + j) ===>a[i][j]元素的值
    void main()
    {    

        int a[3][5];
        int i, j;
        int temp = 0;

        getArray3(a);

        for(i=0; i<3; i++)
        {
            for(j=0; j<5; j++)
            {
                printf("%d ", a[i][j]);
            }
        }



        printf("hello.\n");
        system("pause");
        return;

    }

    等价关系

    数组参数 等效数组参数
    一维数组 char a[30] 一级指针 char *a
    指针数组 char *a[30] 二级指针char **a
    二维数组 char a[5][30] 数组指针变量char (*a)[30]
