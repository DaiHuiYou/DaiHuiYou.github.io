    1 . 选择排序

    // 未优化
template<typename T>
void selectionSort1(T arr[], int n)
{
    for (int i = 0; i < n; i++) {
        for (int j = i + 1; j < n; j++) {
            if (arr[j] < arr[i])
                swap(arr[j], arr[i]);
        }
    }

    return;
}

// 优化
template <typename T>
void selectionSort2(T arr[], int n)
{
    for (int i = 0; i <= n - 1; i++){

        int minIndex = i;
        for (int j = i+1; j <= n - 1; j++) {
            if (arr[minIndex] > arr[j])
                minIndex = j;
        }

        swap(arr[minIndex], arr[i]);
    }
    return;
}
    • 测试

    从 1 万个 0~100 的随机数中，从小到大排序。

    selectionSort1 : 2.208 s

    selectionSort2 : 0.108 s

    2. 插入排序

    // 未优化
template<typename T>
void InsertionSort1(T arr[], int n)
{
    for (int i = 1; i < n; i++)
    {
        for (int j = i; j > 0 && arr[j] < arr[j - 1]; j--) {
            swap(arr[j], arr[j - 1]);
        }
    }    
}

// 优化
template<typename T>
void InsertionSort2(T arr[], int n) {

    for (int i = 1; i < n; i++) {

        T e = arr[i];
        int j;
        for (j = i; j > 0 && arr[j - 1] > e; j--)
            arr[j] = arr[j - 1];
        arr[j] = e;
    }

    return;
}

    从1万个 0~100 的随机数中，从小到大排序。

    3. 希尔排序

    // 希尔排序
template<typename T>
// n 表示数组 capacity
void shellSort(T arr[], int n)
{
    int cnt = n - 1;  // 数组元素个数
    int tmp = 0;       // 临时变量

    // d：distance， 
    for (int d = n/2; d>=1; d=d/2) {
        // 遍历
        for (int i = d; i <=cnt; i++) {
            // 插入排序
            tmp = arr[i];
            for ( int j = i - d; j>=0 && arr[j + d] < arr[j]; j = j - d) {
                arr[j + d] = arr[j];
                arr[j] = tmp;
            }
        }
    }
}

    • 测试

      从 1 万个 0~100 的随机数中，从小到大排序。

      Insertion Sort : 6.332 s

      Shell Sort : 0.029 s

    2. 快速排序

    int __partition(int arr[], int l, int r)
{
    int v = arr[l];

    int j = l;
    for (int i = l + 1; i <= r; i++) {
        if (arr[i] < v) {
            j++;
            swap(arr[i], arr[j]);
        }
    }

    swap(arr[j], arr[l]);
    return j;
}

void __quickSort(int arr[], int l, int r)
{
    if (l >= r)
        return;

    int p = __partition(arr, l, r);
    __quickSort(arr, l, p - 1);
    __quickSort(arr, p + 1, r);
}

// 快速排序
void quickSort(int arr[], int n)
{
    __quickSort(arr, 0, n - 1);
}

    3. 堆排序

    void shiftDown(int arr[], int n, int k)
{
    while (2 * k + 1 < n) {
        int j = 2 * k + 1;
        if (j+1<n && arr[j] < arr[j + 1]) {
            j++;
        }
        if (arr[k] < arr[j]) {
            swap(arr[k], arr[j]);
            k = j;
        }
        else
            break;
    }
}

void heapSort1(int arr[], int n)
{
    // heapify
    for (int i = (n - 1) / 2; i >= 0; i--) {
        shiftDown(arr, n, i);
    }

    for (int i = n - 1; i > 0; i--) {
        swap(arr[0], arr[i]);
        shiftDown(arr, i, 0);
    }
}

// n 为数组元素个数
// k 当前要筛选的索引
// arr[0] 为临时存储变量
void shift(int arr[], int n, int k) 
{
    while (2 * k < n) {
        int j = 2 * k;
        if (j + 1 < n && arr[j] < arr[j + 1])
            j++;
        if (arr[k] < arr[j]) {
            // arr[j] 与 arr[k] 交换
            arr[0] = arr[j];
            arr[j] = arr[k];
            arr[k] = arr[0];

            k = j;
        }
        else 
            break;
    }
}

// n 为数组元素个数
// arr[0] 为临时存储变量
void heapSort2(int arr[], int n)
{
    for (int i = (n - 1) / 2; i >= 1; i--) {
        shift(arr, n, i);
    }

    for (int i = n - 1; i >=2; i--) {
        // arr[i] 与 arr[1] 交换
        arr[0] = arr[i];
        arr[i] = arr[1];
        arr[1] = arr[0];

        shift(arr, i, 1);
    }

}

      从100 万个 0~100 的随机数中，从小到大排序。

      Heap Sort 1 : 2.228 s

      Heap Sort 2 : 0.248 s
